import BrowserOnly from "@docusaurus/BrowserOnly";
import {
    CodeBlockWithAction, SceneComponent,
    sceneDeveloperInteractionManualSelection, sceneDeveloperInteractionTabSelection, sceneFilterPropsInteraction
} from "../../src/components/Scene/SceneComponents";

export const SceneCustomComponent = ({scene, uid}) => {
    return (<BrowserOnly>
        {
            () =>
                <div className='docs__actions'>
                    <CodeBlockWithAction title={'Toggle cameras:'}
                                         buttonText={`Click to toggle CAMERA types`}
                                         scene={scene}
                                         action={`scene.propCtx.toggleSelectedPropType('CAMERA')`}/>
                    <CodeBlockWithAction title={'Search by prop name: '}
                                         buttonText={`Click to search props with '0' in its name`}
                                         scene={scene}
                                         action={`scene.propCtx.searchValue = '0'`}/>
                    <CodeBlockWithAction title={'Reset filter: '}
                                         buttonText={`Click to reset the filter`}
                                         scene={scene}
                                         action={`scene.propCtx.resetFilter()`}/>
                    <CodeBlockWithAction title={'Get a list of all filtered props: '}
                                         buttonText={`Click to get all filtered props`}
                                         scene={scene}
                                         action={`const s = JSON.stringify(scene.propCtx.filteredProps, null, 2)
scene.overlayCtx.openWith(
'<div style="width: 80%; height: 80%; background-color: white; padding: 40px"><textarea style="width: 100%; height: 100%">'+s+'</textarea></div>'
)`}/>
                    <SceneComponent scene={scene} uid={uid} width={'100%'} height={'650px'}/>
                </div>
        }
    </BrowserOnly>)
}

<SceneCustomComponent {...sceneFilterPropsInteraction}/>
